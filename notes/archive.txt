## VM ssh setup ##
sudo apt update -y
sudo apt install openssh-server
sudo systemctl status ssh
q
sudo ufw allow ssh
# enable adapter 2 as a host-only adapter for VB#
ip a
ssh ali@192.168.56.102

## Docker install ##
sudo apt install curl
sudo curl -sSL https://get.docker.com/ | sh
sudo systemctl start docker
docker version

## Add linux user as docker admin ##
sudo groupadd docker
sudo usermod -aG docker $USER
newgrp docker

## MySQL Docker setup ##
docker pull mysql
docker run \
    --name=ali \
    -e MYSQL_ROOT_PASSWORD=12e \
    -p 3306:3306 \
    --restart=unless-stopped \
    -d mysql
docker ps
docker logs ali
docker inspect ali
# private IP address: 172.17.0.2
